Schedule Scraper

A small python program that pulls the HTML code from the online class catalog for Fullerton College. Using Beautiful Soup 4 and Gspread I made an algorithm to pull data from each table element and organize it based on column number and data types. Once the Data is pulled it then sends all the data to a Google Spreadsheet for cloud access at any time.

The purpose behind this program was to make choosing classes easier for me, it filters through classes that are Restricted or waitlisted and only showed those that were either open or closed (so I could see which classes recently closed)

Batch Thumbnail Creator

A program to solve my problem with always having to make thumbnails for websites, photos, and product samples. It takes all images from a folder and resizes them based on the values within a text document. Once resized it puts the newly created pictures in a separate folder.


#Thumbnail.py:
#   Takes in images from folder /Images using OS walk
#   checks sizes from file size.txt
#   resizes images based on the following conditions:
#       Will set width and height to the value
#           If one value is set to auto then constrain to the other value
#           Else resize to both values and stretch
#   Export images to folder /Thumbnails as pngs

import os
from PIL import Image

filePaths = []
fileNames = []
constrain = False
reqWidth = 0
reqHeight = 0
useWidth = False
useHeight = False

def ScanDir():
    print("Pulling file locations from directory: ")
    for root, dirs, files in os.walk(".\Images"):
        print(root)
        print("\n")
        for name in files:
            if not ".db" in name:
                path = os.path.join(root, name)  
                filePaths.append(path)
                fileNames.append(name)
            else:
                continue
        print("Using the following files:")
        for item in filePaths:
            print(item)

def ScanDim():
    count = 0
    f = open("size.txt", 'r')
    for line in f:
        count += 1
        if count == 1:
            colInd = line.find(": ")
            reqWidth = line[colInd+2::]
        elif count == 2:
            colInd = line.find(": ")
            reqHeight = line[colInd+2::]

    print("\n")
    reqWidth, reqHeight, useWidth, useHeight = ConstrainCheck(reqWidth, reqHeight)
    return reqWidth, reqHeight, useWidth, useHeight

def ConstrainCheck(reqWidth, reqHeight):
    if reqWidth == "auto" or reqWidth == "auto\n":
        constrain = True
        useWidth = False
    else:
        reqWidth = int(reqWidth)
        constrain = False
        useWidth = True
        
    if reqHeight == "auto" or reqHeight == "auto\n":
        constrain = True
        useHeight = False
    else:
        reqHeight = int(reqHeight)
        constrain = False
        useHeight = True
        
    print("Set dimensions:")
    print("width: " + str(reqWidth) + ", height: " + str(reqHeight) + " | Constrain:" + str(constrain))
    return reqWidth, reqHeight, useWidth, useHeight

def ResizePropCheck(width, height, reqWidth, reqHeight, useWidth, useHeight):
    if useWidth and useHeight:
        print "both"
        size = reqWidth, reqHeight
        print size
        return size
    elif useWidth and not useHeight:
        print "width only"
        ratio = float(reqWidth)/float(width)
        newHeight = ratio * height
        size = reqWidth, newHeight
        print size
        return size
    elif not useWidth and useHeight:
        print "height only"
        ratio = float(reqHeight)/float(height)
        newWidth = ratio * width
        size = newWidth, height
        print size
        return size
    else:
        print "neither"
    print "useWidth " + str(useWidth)
    print "useHeight " + str(useHeight)

def Resize(reqWidth, reqHeight, useWidth, useHeight):
    outputLoc = "./Thumbnails/"
    count = 0
    for path in filePaths:
        try:
            print path
            im = Image.open(path)
            oriSize = im.size
            size = ResizePropCheck(oriSize[0], oriSize[1], reqWidth, reqHeight, useWidth, useHeight)
            im.thumbnail(size, Image.ANTIALIAS)
            im.save(outputLoc + fileNames[count])
            count += 1
        except:
            print("Exception raised on item:\n" + path)

def main():
    ScanDir()
    reqWidth, reqHeight, useWidth, useHeight = ScanDim()
    Resize(reqWidth, reqHeight, useWidth, useHeight)

if __name__ == "__main__":
    main()
